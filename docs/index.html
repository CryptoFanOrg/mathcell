MathCell is a simple straightforward way to include interactive mathematics in a web page. First copy this template into the body of the page:

<div class="mathcell" style="height: 4in">
<script>

var parent = document.scripts[ document.scripts.length - 1 ].parentNode;

var id = 'id' + document.scripts.length;
parent.id = id;

MathCell( id, [ { type: '' } ] );

parent.update = function( id ) {

  var x = +document.getElementById( id + 'x' ).value;

  var data = [];

  var config = { type: '' };

  evaluate( id, data, config );

}

parent.update( id );

</script>
</div>

Interactive elements are specified in the dictionary inside the MathCell() command. The output elment is specified in the config dictionary inside the update function.
